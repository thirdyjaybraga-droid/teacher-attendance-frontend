<script>

const params = new URLSearchParams(location.search);

const id = params.get("id");



// TODO: replace API variable with your backend URL (no trailing slash)

const API = "https://teacher-attendance-backend-vbzs.onrender.com"; // e.g. https://teacher-attendance-backend.onrender.com

const DASHBOARD = "https://thirdyjaybraga-droid.github.io/teacher-attendance/dashboard.html";



document.getElementById("to-dashboard").href = DASHBOARD;



if (!id) {

document.getElementById("status").textContent = "No ID provided";

document.getElementById("message").textContent = "Scan a valid QR code.";

} else {

fetch(API + "/api/teachers")

.then(r => r.json())

.then(list => {

const t = list.find(x => x.id === id);

if (t) {

document.getElementById("status").textContent = "Welcome";

document.getElementById("teacher-name").textContent = t.name;

document.getElementById("message").textContent = "Your attendance has been recorded. Thanks!";

document.querySelector(".card").classList.add("ok");

} else {

document.getElementById("status").textContent = "Unknown ID";

document.getElementById("message").textContent = "Teacher not found.";

}

})

.catch(err => {

document.getElementById("status").textContent = "Error";

document.getElementById("message").textContent = "Could not contact server.";

console.error(err);

});



// after 5s, optionally go to dashboard

setTimeout(()=> {

window.location.href = DASHBOARD;

}, 5000);

}

</script>

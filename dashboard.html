<script>

const API = "https://teacher-attendance-backend-vbzs.onrender.com"; // replace



async function load(){

try {

const res = await fetch(API + "/api/attendance");

const rows = await res.json();

const tbody = document.getElementById("body");

tbody.innerHTML = "";

rows.forEach(r => {

const tr = document.createElement("tr");

tr.innerHTML = `<td>${r.id}</td><td>${r.name}</td><td class="${r.status==='Present'?'present':''}">${r.status}</td><td>${r.timestamp}</td>`;

tbody.appendChild(tr);

});

} catch(err){

console.error(err);

}

}

document.getElementById("download").onclick = () => {

window.open(API + "/download", "_blank");

}

load();

setInterval(load, 5000);

</script>
